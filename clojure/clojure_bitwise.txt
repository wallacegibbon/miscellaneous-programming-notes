;; make some test with 0xaa, which is 170 in decimal

;;-----------------------------------------------------------------------------
;; the and, or, xor, not
;;-----------------------------------------------------------------------------
(bit-and 0xaa 0x07)
;> 2

(bit-or 0xaa 0x07)
;> 175

(bit-xor 0xaa 0x07)
;> 173

(bit-not 0xaa)
;> -171


;;-----------------------------------------------------------------------------
;; these functions accept a number and a index:
;;  bit-flip, bit-set, bit-clear
;;  bit-shift-left, bit-shift-right
;;-----------------------------------------------------------------------------
(bit-flip 0xaa 0)
;> 171

(bit-set 0xaa 0)
;> 171

(bit-clear 0xaa 1)
;> 168


;;-----------------------------------------------------------------------------
;; bit-test check bit on index, return true when 1, false when 0
;;-----------------------------------------------------------------------------
(bit-test 0xaa 0)
;> false

(bit-test 0xaa 1)
;> true


;;-----------------------------------------------------------------------------
;; the shift
;;-----------------------------------------------------------------------------
(bit-shift-left 1 2)
;> 4

(bit-shift-right 8 2)
;> 2
